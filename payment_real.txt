AccessOrchestra — Payment Architecture Deep Dive
The Core Concept
Each uAgent has its own crypto wallet derived from its seed phrase. When a student's request flows through the system, the Orchestrator pays each specialist agent in FET tokens (Fetch.ai's native currency) for the work it performs. This transforms AccessOrchestra from a chatbot into an economic marketplace — the key differentiator for judges.

The 5 Agents & What They Earn Per Session
Agent	Role for Disabled Students	Payment	Type
Adaptive Format Agent	Generates all 4 accessibility formats simultaneously (Easy Read, Caption, Screen Reader, Chunked)	0.003 FET	REAL on-chain
Speech Transcription Agent	Converts professor's speech to text in real-time	0.002 FET	Mocked
Language Simplification Agent	Simplifies complex STEM language	0.002 FET	Mocked

The strategy is deliberate: one real on-chain payment wins the demo, four mocked payments tell the full marketplace story.

The 3 Technical Components
1. fund_agent_if_low() Called once at agent startup. Automatically tops up each agent's wallet from the testnet faucet so the wallet never drains mid-demo.

@agent.on_event("startup")
async def startup(ctx: Context):
    fund_agent_if_low(agent.wallet.address())
2. ctx.send_money() Sends atestfet (testnet FET) from Orchestrator to a specialist agent's wallet. Returns a transaction hash. The denomination conversion is:

1 FET = 10^18 atestfet
0.003 FET = 30,000 atestfet (in simplified hackathon units)
3. ctx.ledger.query_bank_balance() Checks on-chain wallet balance. Used for pre-demo verification and by receiving agents to confirm payment arrived.

The Critical Async Pattern
This is the most important technical insight in the document — do NOT await payment before sending the work request:

# WRONG — blocks the pipeline for 5-15 seconds
tx = await ctx.send_money(dest, amount)
await wait_for_tx_to_complete(tx)   # pipeline stalls
await ctx.send(dest, FormatRequest(...))
# CORRECT — payment and work request fire concurrently
tx_task = asyncio.create_task(ctx.send_money(dest, amount))
work_task = asyncio.create_task(ctx.send(dest, FormatRequest(...)))
tx_hash, _ = await asyncio.gather(tx_task, work_task)
Testnet confirmation takes 5–15 seconds. If you wait for it synchronously, the demo dies. The tx_hash is logged as proof and verified separately.

import os
from datetime import datetime, timezone
from uuid import uuid4
from cosmpy.aerial.client import LedgerClient, NetworkConfig
from uagents import Context, Protocol
from uagents_core.contrib.protocols.chat import ChatMessage, EndSessionContent, TextContent
from uagents_core.contrib.protocols.payment import (
    CancelPayment, CommitPayment, CompletePayment,
    Funds, RejectPayment, RequestPayment,
    payment_protocol_spec,
)
# ↓ swap these for your specialist agent addresses
from your_project.config import CALENDAR_ADDRESS, FOCUS_ADDRESS, TASK_ADDRESS
from your_project.models import SharedAgentState
from your_project.state_service import state_service
payment_proto = Protocol(spec=payment_protocol_spec, role="seller")
_orchestrator_wallet = None   # injected at startup — do not change this line
FET_AMOUNT = os.getenv("FIXED_FET_AMOUNT", "0.1")
FET_FUNDS  = Funds(currency="FET", amount=FET_AMOUNT, payment_method="fet_direct")
def set_orchestrator_wallet(wallet) -> None:
    global _orchestrator_wallet
    _orchestrator_wallet = wallet
# ─── STEP A: send payment request to user ────────────────────────────────────
async def request_payment_from_user(ctx: Context, user_address: str, description: str) -> None:
    use_testnet = os.getenv("AGENT_NETWORK", "testnet") == "testnet"
    recipient   = str(_orchestrator_wallet.address())
    metadata = {
        "fet_network": "stable-testnet" if use_testnet else "mainnet",
        "mainnet":     "false"          if use_testnet else "true",
        "content":     description,   # ← ASI:One renders this inside the payment card
        "provider_agent_wallet": recipient,
    }
    await ctx.send(user_address, RequestPayment(
        accepted_funds=[FET_FUNDS],
        recipient=recipient,          # orchestrator's fetch1... wallet
        deadline_seconds=300,
        reference=str(ctx.session),
        description=description,
        metadata=metadata,
    ))
# ─── STEP B: user approved → verify on-chain → do the work ──────────────────
@payment_proto.on_message(CommitPayment)
async def handle_commit_payment(ctx: Context, sender: str, msg: CommitPayment):
    buyer_wallet = None
    if isinstance(msg.metadata, dict):
        buyer_wallet = (msg.metadata.get("buyer_fet_wallet")
                        or msg.metadata.get("buyer_fet_address"))
    verified = (
        msg.funds.payment_method == "fet_direct"
        and buyer_wallet
        and _verify_payment(msg.transaction_id, str(msg.funds.amount), buyer_wallet)
    )
    if not verified:
        await ctx.send(sender, CancelPayment(
            transaction_id=msg.transaction_id,
            reason="On-chain verification failed.",
        ))
        return
    await ctx.send(sender, CompletePayment(transaction_id=msg.transaction_id))
    # Retrieve what you stored in the chat handler
    routing        = ctx.storage.get(f"routing:{sender}") or ""
    prompt         = ctx.storage.get(f"prompt:{sender}")  or ""
    session_id     = ctx.storage.get(f"session:{sender}") or msg.transaction_id
    state = SharedAgentState(
        chat_session_id=session_id,
        query=prompt,
        user_sender_address=sender,
        payment_tx=msg.transaction_id,   # ← attach verified TX to state
    )
    state_service.set_state(session_id, state)
    # ↓ swap routing keys and addresses for your agents
    if "calendar" in routing:
        await ctx.send(CALENDAR_ADDRESS, state)
    elif "focus" in routing:
        await ctx.send(FOCUS_ADDRESS, state)
    elif "task" in routing:
        await ctx.send(TASK_ADDRESS, state)
    ctx.storage.remove(f"routing:{sender}")
    ctx.storage.remove(f"prompt:{sender}")
    ctx.storage.remove(f"session:{sender}")
@payment_proto.on_message(RejectPayment)
async def handle_reject_payment(ctx: Context, sender: str, msg: RejectPayment):
    ctx.storage.remove(f"routing:{sender}")
    ctx.storage.remove(f"prompt:{sender}")
    ctx.storage.remove(f"session:{sender}")
    await ctx.send(sender, ChatMessage(
        timestamp=datetime.now(tz=timezone.utc), msg_id=uuid4(),
        content=[
            TextContent(type="text", text="Payment declined. Send your message again when ready."),
            EndSessionContent(type="end-session"),
        ],
    ))
# ─── on-chain verification (no changes needed) ───────────────────────────────
def _verify_payment(transaction_id: str, expected_fet: str, sender_wallet: str) -> bool:
    try:
        use_testnet = os.getenv("AGENT_NETWORK", "testnet") == "testnet"
        ledger = LedgerClient(
            NetworkConfig.fetchai_stable_testnet() if use_testnet
            else NetworkConfig.fetchai_mainnet()
        )
        denom            = "atestfet" if use_testnet else "afet"
        expected_amount  = int(float(expected_fet) * 10**18)
        expected_recipient = str(_orchestrator_wallet.address())
        tx = ledger.query_tx(transaction_id)
        if not tx.is_successful():
            return False
        for event_type, attrs in tx.events.items():
            if event_type == "transfer":
                if (attrs.get("recipient") == expected_recipient
                        and attrs.get("sender") == sender_wallet):
                    amount_str = attrs.get("amount", "")
                    if amount_str.endswith(denom):
                        paid = int(amount_str.replace(denom, ""))
                        if paid >= expected_amount:
                            return True
        return False
    except Exception:
        return False
